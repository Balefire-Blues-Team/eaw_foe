scripted_gui = {
	geck_decision_category_scripted_gui = {
		context_type = decision_category
		
		window_name = "geck_decision_category_scripted_gui"
		
		effects = {
			entry_button_click = {
				ROOT = {
					if = {
						limit = {
							check_variable = { geck_target_state = PREV.id }
						}
						clear_variable = geck_target_state
					}
					else = {
						set_variable = { geck_target_state = PREV.id }
					}
				}
			}
		}
		
		triggers = {
			entry_checkmark_visible = {
				ROOT = {
					check_variable = { geck_target_state = PREV.id }
				}
			}
		}
		
		dynamic_lists = {
			geck_decision_category_scripted_gui_gridbox = {
				array = owned_states
				entry_container = geck_decision_category_scripted_gui_entry
				change_scope = yes
			}
		}
	}
}
