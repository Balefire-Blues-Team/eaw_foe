scripted_gui = {
	converted_state_ncn = {

		context_type = selected_state_context

		window_name = "converted_state_container_ncn"
		parent_window_token = selected_state_view

		visible = {
			ROOT = { tag = NCN }
			OR = {
				has_state_flag = ncn_converted
				is_owned_and_controlled_by = NCN
			}
			NOT = { has_state_flag = is_impassable }
		}

		effects = {}
	}

	converted_state_hea = {

		context_type = selected_state_context

		window_name = "converted_state_container_hea"
		parent_window_token = selected_state_view

		visible = {
			ROOT = { tag = HEA }
			OR = {
				has_state_flag = hea_converted
				is_owned_and_controlled_by = HEA
			}
			NOT = { has_state_flag = is_impassable }
		}

		effects = {}
	}

	state_conversion_button_ncn = {
		context_type = selected_state_context

		window_name = "convertable_state_container_ncn"
		parent_window_token = selected_state_view

		visible = {
			ROOT = { tag = NCN }
			NOT = { is_owned_and_controlled_by = NCN }
			NOT = { has_state_flag = ncn_converted }
			AND = {
				any_neighbor_state = {
					is_owned_and_controlled_by = NCN
				}
				can_be_converted_NCN = yes
			}
			NOT = { has_state_flag = is_impassable }
		}

		effects = {}
	}

	state_conversion_button_hea = {
		context_type = selected_state_context

		window_name = "convertable_state_container_hea"
		parent_window_token = selected_state_view

		visible = {
			ROOT = { tag = HEA }
			NOT = { is_owned_and_controlled_by = HEA }
			NOT = { has_state_flag = hea_converted }
			AND = {
				any_neighbor_state = {
					is_owned_and_controlled_by = HEA
				}
				can_be_converted_HEA = yes
			}
			NOT = { has_state_flag = is_impassable }
		}

		effects = {}
	}
}
