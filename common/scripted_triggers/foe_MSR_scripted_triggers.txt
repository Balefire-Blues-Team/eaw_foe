#Country scope
is_MATO_member = { 
	custom_trigger_tooltip = {
		tooltip = is_MATO_member_tt
		OR = {
			has_country_flag = MATO_member_flag
			AND = {
				is_in_faction_with = MSR
				has_global_flag = MATO_formed_flag
			}
		}
	}
}
