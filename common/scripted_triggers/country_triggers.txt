is_mormon_nation = {
	custom_trigger_tooltip = {
		tooltip = is_mormon_nation_tt
		OR = {
			original_tag = NCN
			has_country_flag = is_mormon_nation
		}
		NOT = { has_country_flag = not_a_mormon_nation }
	}
}

is_not_mormon_nation = {
	custom_trigger_tooltip = {
		tooltip = is_not_mormon_nation_tt
		is_mormon_nation = no
	}
}
