is_tribal_nation_allowed_block = {
	custom_trigger_tooltip = {
		tooltip = is_tribal_nation_tt
		OR = {
			has_country_flag = is_tribal_nation

			#Generic Trees
			tag = FRT
			tag = EVR
			tag = BOG
			tag = HSC
			tag = HSW

			#Auxiliary Trees
			tag = HIT
			tag = GRV
			tag = FTP
			tag = FTR
			tag = WNF
		}
	}
}

	}

	}
