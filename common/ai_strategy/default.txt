ai_army_building_minimum_puppet = { #Minimum Army Size Enforced
	enable = {
		has_army_size = { size < 10 }
		has_war = no
		is_subject = yes
	}
	abort = {
		OR = {
			NOT = { has_army_size = { size < 10 } }
			has_war = yes
		}
	}
	ai_strategy = {
		type = build_army
		id = infantry
		value = 100
	}
}

stop_building_too_many_units = { # Please Stop Building Units
	enable = {
		AND = {
			NOT = { ai_wants_divisions < 300 }
			has_army_size = { size > 300 }
			num_of_factories > 99
		}
	}
	abort = {
		OR = {
			ai_wants_divisions < 300
			has_army_size = { size < 300 }
			num_of_factories < 99
		}
	}
	ai_strategy = {
		type = role_ratio
		id = infantry
		value = -500

		type = role_ratio

		type = role_ratio
		id = armor
		value = -500
	}
	ai_strategy = {
		type = role_ratio
		id = combat_robot
		value = -500
	}
	ai_strategy = {
		type = role_ratio
		id = support_robot
		value = -500
	}
}

save_admiral_pp_for_later = {
	enable = {
		has_war = no
		threat < 0.3
		has_political_power < 200
	}
	abort = {
		OR = {
			has_war = yes
			threat > 0.3
			has_political_power > 200
		}
	}

	ai_strategy = { # dont burn PP on admirals (can still get free ones)
		type = pp_spend_priority
		id = admiral
		value = -100
	}
}
