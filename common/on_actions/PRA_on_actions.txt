on_actions = {
    on_startup = {
        effect = {
            PRA = {

                    set_variable = { Raider_Power = 50};
                    set_variable = { Guard_Power = 1};
                    set_variable = { Church_Power = 1};

                    set_variable = { Raider_Pony_Power = .8};
                    set_variable = { Guard_Pony_Power = .1};
                    set_variable = { Church_Pony_Power = .1};

                    set_variable = { CF_Alligency = 0 };

                    set_variable = { Raider_Power_Change = 0};
                    set_variable = { Guard_Power_Change = 0};
                    set_variable = { Church_Power_Change = 0};

                    set_variable = { Raider_Pony_Power_Change = 0};
                    set_variable = { Guard_Pony_Power_Change = 0};
                    set_variable = { Church_Pony_Power_Change = 0};

                    set_variable = { ChainAlig = 3 };
                }
            }
        }

    on_daily_PRA ={
        effect = {

            if = {
                limit = { NOT = { has_country_flag = PRA_Civil_War_Started } }

                add_to_variable = {
                    var = Raider_Power
                    value = Raider_Power_Change
                    }

                add_to_variable = {
                    var = Guard_Power
                    value = Guard_Power_Change
                }
                add_to_variable = {
                    var = Church_Power
                    value = Church_Power_Change
                }

                add_to_variable = {
                    var = Raider_Pony_Power
                    value = Raider_Pony_Power_Change
                }
                add_to_variable = {
                    var = Guard_Pony_Power
                    value = Guard_Pony_Power_Change
                }
                add_to_variable = {
                    var = Church_Pony_Power
                    value = Church_Pony_Power_Change
                }

            }
        }
    }
