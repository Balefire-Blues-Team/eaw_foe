
AOS_increase_loyalty = {
	IF = {
		limit = { 
			AOS = {
				check_variable = {
					AOS_khan_loyalty < 10
				} 
			}
		}
		custom_effect_tooltip = KHA_AOS_increase_loyalty_tt
		AOS = {
			add_to_variable = {
				var = AOS_khan_loyalty
				value = 1
			}
		}
	}
	ELSE_IF = {
		limit = { 
			AOS = {
				check_variable = {
					AOS_khan_loyalty > 9
				} 
			}
		}
		add_political_power = 20
		custom_effect_tooltip = KHA_increase_influence_2_tt
		add_to_variable = {
			var = KHA_khan_influence
			value = 2
		}
	}
}
