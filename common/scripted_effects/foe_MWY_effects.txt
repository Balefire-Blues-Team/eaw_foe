# Raise Levy

	MWY_raise_grainlands_levy_effect = {
		if = {
			limit = { controls_state = 553 }
			if = {
				limit = { has_country_flag = marshal_train_levies_grainlands_flag }
				if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 2 } }
					load_oob = MWY_grainlands_trained_levy_oob_1
				}
				else_if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 3 } }
					load_oob = MWY_grainlands_trained_levy_oob_2
				}
				else_if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 4 } }
					load_oob = MWY_grainlands_trained_levy_oob_3
				}
				else_if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 5 } }
					load_oob = MWY_grainlands_trained_levy_oob_4
				}
				else_if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 6 } }
					load_oob = MWY_grainlands_trained_levy_oob_5
				}
			}
			else = {
				if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 2 } }
					load_oob = MWY_grainlands_levy_oob_1
				}
				else_if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 3 } }
					load_oob = MWY_grainlands_levy_oob_2
				}
				else_if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 4 } }
					load_oob = MWY_grainlands_levy_oob_3
				}
				else_if = {
					limit = { check_variable = { mwy_grainlands_levy_var = 5 } }
					load_oob = MWY_grainlands_levy_oob_4
				}

	}

		set_variable = { realm_poor_opinion_count_var = 0 }
		if = {
			limit = {
				check_variable = { mwy_grainlands_opinion = 1 }

			add_to_variable = { realm_poor_opinion_count_var = 1 }
		}
		if = {
			limit = {
				check_variable = { mwy_amethyst_opinion = 1 }
			}
			add_to_variable = { realm_poor_opinion_count_var = 1 }
