
UNI_mind_influences = {
	hidden_effect = {
		if = {
			limit = {
				tag = UNI
				continuation > 0.14
				continuation < 0.30
			}	
			if = {
				limit = {
					NOT = { has_idea = UNI_twilight_influence }
					NOT = { has_idea = UNI_small_twilight_influence }
				}
				add_ideas = UNI_small_twilight_influence
			}
		}

		if = {
			limit = {
				tag = UNI
				continuation > 0.14
				continuation < 0.30
			}	
			if = {
				limit = {
					has_idea = UNI_twilight_influence
					NOT = { has_idea = UNI_small_twilight_influence }
				}
				swap_ideas = {
					add_idea = UNI_small_twilight_influence
					remove_idea = UNI_twilight_influence
				}
			}
		}
		if = {
			limit = {
				tag = UNI
				continuation < 0.15
			}	
			if = {
				limit = {
					has_idea = UNI_small_twilight_influence 
				}
				remove_ideas = UNI_small_twilight_influence
			}
		}
		if = {
			limit = {
				tag = UNI
				continuation > 0.29
			}	
			if = {
				limit = {
					NOT = { has_idea = UNI_twilight_influence }
				}
				add_ideas = UNI_twilight_influence
			}
		}
		if = {
			limit = {
				tag = UNI
				continuation < 0.30
			}	
			if = {
				limit = {
					has_idea = UNI_twilight_influence 
				}
				remove_ideas = UNI_twilight_influence
			}
		}
		if = {
			limit = {
				tag = UNI
				survivalism > 0.29
			}	
			if = {
				limit = {
					NOT = { has_idea = UNI_GM_influence }
				}
				add_ideas = UNI_GM_influence
			}
		}
		if = {
			limit = {
				tag = UNI
				survivalism < 0.30
			}	
			if = {
				limit = {
					has_idea = UNI_GM_influence
				}
				remove_ideas = UNI_GM_influence
			}
		}
		if = {
			limit = {
				tag = UNI
				survivalism > 0.14
				survivalism < 0.30
			}	
			if = {
				limit = {
					NOT = { has_idea = UNI_GM_influence }
					NOT = { has_idea = UNI_small_GM_influence }
				}
				add_ideas = UNI_small_GM_influence
			}
		}

		if = {
			limit = {
				tag = UNI
				survivalism > 0.14
				survivalism < 0.30
			}	
			if = {
				limit = {
					has_idea = UNI_GM_influence
					NOT = { has_idea = UNI_small_GM_influence }
				}
				swap_ideas = {
					add_idea = UNI_small_GM_influence
					remove_idea = UNI_GM_influence
				}
			}
		}
		if = {
			limit = {
				tag = UNI
				survivalism < 0.15
			}	
			if = {
				limit = {
					has_idea = UNI_small_GM_influence 
				}
				remove_ideas = UNI_small_GM_influence
			}
		}

	}
}

UNI_raise_vat_contruction_frame = {
	if = {
		limit = { has_country_flag = UNI_vatframe_1_flag }
		clr_country_flag = UNI_vatframe_1_flag
		set_country_flag = UNI_vatframe_2_flag
	}
	else_if = {
		limit = { has_country_flag = UNI_vatframe_2_flag }
		clr_country_flag = UNI_vatframe_2_flag
		set_country_flag = UNI_vatframe_3_flag
	}
	else_if = {
		limit = { has_country_flag = UNI_vatframe_3_flag }
		clr_country_flag = UNI_vatframe_3_flag
		set_country_flag = UNI_vatframe_4_flag
	}
	else = { set_country_flag = UNI_vatframe_1_flag }
}

UNI_raise_vat_contruction_plating = {
	if = {
		limit = { has_country_flag = UNI_vatplating_1_flag }
		clr_country_flag = UNI_vatplating_1_flag
		set_country_flag = UNI_vatplating_2_flag
	}
	else = { set_country_flag = UNI_vatplating_1_flag }
}


UNI_finish_male_research = {
	if = {
		limit = { 
			check_variable = { male_research > 99 } 
		}
		set_country_flag = UNI_Male_alicorn_finished
	}
}
