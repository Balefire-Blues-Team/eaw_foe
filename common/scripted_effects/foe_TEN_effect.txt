		limit = { has_country_flag = TEN_penal_1_flag }

	}
	else_if = {
		limit = { has_country_flag = TEN_penal_2_flag }
		clr_country_flag = TEN_penal_2_flag
		set_country_flag = TEN_penal_3_flag
	}

		set_country_flag = TEN_penal_4_flag

}
