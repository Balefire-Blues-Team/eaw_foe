set_unit_leader_to_raider = {
	if = {
		limit = {
			is_female = yes
		}
		randomize_variable = {
			var = name_idx
			distribution = uniform
			min = 1
			max = 29
		}
		round_variable = name_idx
		randomize_variable = {
			var = surname_idx
			distribution = uniform
			min = 1
			max = 24
		}
		round_variable = surname_idx
		meta_effect = {
			text = {
				set_leader_name = "[NAME] [SURNAME]"
			}
			NAME = "[?name_idx.GetRaiderFemaleName]"
			SURNAME = "[?surname_idx.GetRaiderSurname]"
		}
		clear_variable = name_idx
		clear_variable = surname_idx
		if = {
			limit = {
				is_navy_leader = yes
			}
			randomize_variable = {
				var = name_idx
				distribution = uniform
				min = 1
				max = 4
			}
			round_variable = name_idx
			meta_effect = {
				text = {
					set_leader_portrait = "[PORTRAIT]"
				}
				PORTRAIT = "[?name_idx.Get_raider_female_portrait]"
			}
			clear_variable = name_idx
		}
		else = {
			randomize_variable = {
				var = name_idx
				distribution = uniform
				min = 1
				max = 4
			}
			round_variable = name_idx
			meta_effect = {
				text = {
					set_leader_portrait = "[PORTRAIT]"
				}
				PORTRAIT = "[?name_idx.Get_raider_female_portrait]"
			}
			clear_variable = name_idx
		}
	}
	else = {
		randomize_variable = {
			var = name_idx
			distribution = uniform
			min = 1
			max = 25
		}
		round_variable = name_idx
		randomize_variable = {
			var = surname_idx
			distribution = uniform
			min = 1
			max = 24
		}
		round_variable = surname_idx
		meta_effect = {
			text = {
				set_leader_name = "[NAME] [SURNAME]"
			}
			NAME = "[?name_idx.GetRaiderMaleName]"
			SURNAME = "[?surname_idx.GetRaiderSurname]"
		}
		clear_variable = name_idx
		clear_variable = surname_idx
		if = {
			limit = {
				is_navy_leader = yes
			}
			randomize_variable = {
				var = name_idx
				distribution = uniform
				min = 1
				max = 4
			}
			round_variable = name_idx
			meta_effect = {
				text = {
					set_leader_portrait = "[PORTRAIT]"
				}
				PORTRAIT = "[?name_idx.Get_raider_male_portrait]"
			}
			clear_variable = name_idx
		}
		else = {
			randomize_variable = {
				var = name_idx
				distribution = uniform
				min = 1
				max = 4
			}
			round_variable = name_idx
			meta_effect = {
				text = {
					set_leader_portrait = "[PORTRAIT]"
				}
				PORTRAIT = "[?name_idx.Get_raider_male_portrait]"
			}
			clear_variable = name_idx
		}
	}
	add_unit_leader_trait = raider_commander
}
