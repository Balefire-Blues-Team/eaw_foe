focus = { 
		id = GR_factorytown
		icon = GFX_goal_generic_military_industry_focus
		relative_position_id = GR_take_control
		x = -1
		y = 1
		cost = 1

		available_if_capitulated = yes 
		prerequisite = { focus = GR_take_control } 

		mutually_exclusive = { 
        focus = GR_raidertown
        }

		ai_will_do = { 
			factor = 100
		} 

		completion_reward = {
			log = "[GetDateText]: [Root.GetName]: Focus GR_factorytown"
			
			random_owned_controlled_state = {
			limit = {
				free_building_slots = {
					building = arms_factory
					size > 0
					include_locked = yes
				}
				OR = {
					is_in_home_area = yes
					NOT = {
						owner = {
							any_owned_state = {
								free_building_slots = {
									building = arms_factory
									size > 0
									include_locked = yes
								}
								is_in_home_area = yes
							}
						}
					}
				}
			}
			add_extra_state_shared_building_slots = 1
			add_building_construction = {
				type = arms_factory
				level = 1
				instant_build = yes
			}
		}

	}
}
