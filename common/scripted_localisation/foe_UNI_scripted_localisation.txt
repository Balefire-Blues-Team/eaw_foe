# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = UNIGetREDTrust
	text = {
		trigger = { 
			check_variable = {
                UNI_RED_trust < 0
            }
		}
		localization_key = UNI_no_trust_key
	}
	text = {
		trigger = { 
			check_variable = {
                UNI_RED_trust > 0
            }
			check_variable = {
                UNI_RED_trust < 4
            }
		}
		localization_key = UNI_smol_trust_key
	}
	text = {
		trigger = { 
			check_variable = {
                UNI_RED_trust > 3
            }
			check_variable = {
                UNI_RED_trust < 7
            }
		}
		localization_key = UNI_bit_trust_key
	}
	text = {
		trigger = { 
			check_variable = {
                UNI_RED_trust > 6
            }
			check_variable = {
                UNI_RED_trust < 11
            }
		}
		localization_key = UNI_med_trust_key
	}
	text = {
		trigger = { 
			check_variable = {
                UNI_RED_trust > 10
            }
			check_variable = {
                UNI_RED_trust < 15
            }
		}
		localization_key = UNI_high_trust_key
	}
	text = {
		trigger = { 
			check_variable = {
                UNI_RED_trust > 14
            }
		}
		localization_key = UNI_total_trust_key
	}
}
