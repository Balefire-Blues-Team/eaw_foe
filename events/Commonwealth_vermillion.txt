country_event = {
 	id = vermillionists.500
 	title = vermillionists.500.t
 	desc = vermillionists.500.d
	picture = GFX_event_test

	trigger = {
		tag = COR
		COR = {
			exists = yes
			has_country_flag = COR_owes_stratusburg_to_NOV_flag
		}
		any_state = {
			AND = {
				is_core_of = DSB
				is_owned_by = COR
			}
		}
	}
	hidden = yes
 	fire_only_once = yes

	option = {
		name = vermillionists.500.a
		ai_chance = { factor = 50}
		IF = {
			limit = { NOV = {exists = no} }
			COR = {release = NOV}
		}
		every_state = {
			limit = {
				AND = {
					is_core_of = DSB
					is_owned_by = COR
				}
			}
			transfer_state_to = NOV
		}
		COR = {
			clr_country_flag = COR_owes_stratusburg_to_NOV_flag
		}
	}
}
