country_event = {
	id = scavenge.1
	title = scavenge.1.t
	desc = scavenge.1.d
	picture = GFX_event_generic_raider_1

	is_triggered_only = yes

	immediate = {
		set_variable = { resolve_rng = random }
	}

	option = { # Send out a search party
		name = scavenge.1.a
		trigger = {
			command_power > 9
		}
		add_command_power = -10
		random_list = {
			25 = { } # No effect
			25 = { add_manpower = 6 }
			25 = { add_manpower = 11 }
			25 = { add_manpower = 17 }
		}
		clr_country_flag = is_scavenging
	}

	option = { # They are gonners
		name = scavenge.1.b
		clr_country_flag = is_scavenging
		add_political_power = -10
		add_stability = -0.02
	}
}
