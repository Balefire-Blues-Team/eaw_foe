			factor = 5000
		}

   option = { #Now's the time for betrayal
	   name = khanate.11.c
	   	trigger = {
			OR = {
				AND = {
					tag = AOS
					check_variable = {
						AOS_khan_loyalty < 4
					} 
				}
				AND = {
					tag = ARR
					check_variable = {
						ARR_khan_loyalty < 4
					} 
				}
				AND = {
					tag = DEC
					check_variable = {
						DEC_khan_loyalty < 4
					} 
				}
			}
	   	}
		ai_chance = {
			factor = 500
		}
		set_country_flag = KHA_violent_betrayal_flag
		leave_faction = yes
		KHA = {
			ROOT = {
				grant_wargoals_on_core_states_of_prev = yes
			}
		}

			KHA = {
				grant_wargoals_on_core_states_of_prev = yes
			}
