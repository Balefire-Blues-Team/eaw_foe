add_namespace = special_rulers

#Initiate Ruler choice
country_event = {
	id = special_rulers.1
	title = special_rulers.1.t
	desc = special_rulers.1.d
	picture = GFX_event_NCR_startup
	is_triggered_only = yes
	fire_only_once = no

	option = {
		name = special_rulers.1.a
		ai_chance = { 
			factor = 6
			modifier = {
				add = -3
				is_major = yes
			}
			modifier = {
				add = -1
				num_of_civilian_factories > 7
			}
			modifier = {
				add = -1
				num_of_military_factories > 7
			}
			modifier = {
				add = -4
				any_neighbor_country = {
					is_ai = no
				}
			}
		}
		country_event = { id = special_rulers.2}
		set_country_flag = hyper_aggressive_ai
	}
	option = {
		name = special_rulers.1.b
		ai_chance = { factor = 80}
	}
}
