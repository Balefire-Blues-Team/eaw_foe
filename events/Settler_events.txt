#What we did
country_event = {
 	id = settlers.2
 	title = settlers.2.t
 	desc = settlers.2.d
	picture = GFX_event_test

	is_triggered_only = yes
 	fire_only_once = no
 	
 	
	option = {
		name = settlers.2.a
		ai_chance = { factor = 40}
		 
			random_owned_controlled_state = {
				limit = { is_capital = yes }
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = industrial_complex
					level = 1
					instant_build = yes
				}
			}
	}

	option = {
		name = settlers.2.b
		ai_chance = { factor = 30}
		
			random_owned_controlled_state = {
				limit = { is_capital = yes }
				add_extra_state_shared_building_slots = 2
				add_manpower = 500
			}
	}
	
	option = {
		name = settlers.2.c
		ai_chance = { factor = 30}
			division_template = {
	                name = "Capital Guard"
	                is_locked = no
	                    regiments = {
		                    infantry = { x = 0 y = 0 }
							infantry = { x = 0 y = 1 }
							infantry = { x = 1 y = 0 }
							infantry = { x = 1 y = 1 }
							infantry = { x = 0 y = 2 }
							infantry = { x = 1 y = 2 }
	                    }
                    }
			random_owned_controlled_state = {
				limit = { is_capital = yes }
					create_unit = {
						division = "name = \"Stalward Defenders\" division_template = \"Capital Guard\" start_experience_factor = 0.2 start_equipment_factor = 1.0"
						owner = ROOT
					}
			}
	}
}
