				check_variable = {

					NCG_build_infl = NCG_mine_infl
				}
			}
			OR = {
				check_variable = {
					NCG_build_infl > NCG_mech_infl
				}
				check_variable = {
					NCG_build_infl = NCG_mech_infl
				}
			}
		}
		name = neocastle.10.a
		set_country_flag = NCG_unlock_corpo_tree_flag
		load_focus_tree = NCG_focus
	}
	option = {
		trigger = {
			OR = {
				check_variable = {
					NCG_mine_infl = NCG_mech_infl
				}
				check_variable = {
					NCG_mine_infl > NCG_mech_infl
				}
			}
			check_variable = {
                NCG_mine_infl > NCG_build_infl
            }
		}
		name = neocastle.10.b
