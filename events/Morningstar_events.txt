country_event = {
	id = goldendawnrevolt.666
	immediate = {log = "[GetDateText]: [Root.GetName]: event goldendawnrevolt.666"}
	hidden = yes

	mean_time_to_happen = {
		days = 1
	}

	trigger = {
		tag = TMS
		has_completed_focus = TMS_trial_of_blood
		any_neighbor_country = {
			NOT = {
				OR = {
					has_war_with = ROOT 
					is_in_faction_with = ROOT
				}
			}
		}
	}

	option = {
		every_neighbor_country = {
			limit = {
				NOT = {
					OR = {
						has_war_with = ROOT
						is_in_faction_with = ROOT
					}
				}
			}
			ROOT = { 
				hidden_effect = {
					create_wargoal = { 
						type = annex_everything
						target = PREV
					}
				}
				declare_war_on = { target = PREV type = annex_everything } 
			}
		}
	}

}
